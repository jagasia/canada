package com.canadalife.demo.config;

import io.swagger.v3.oas.annotations.OpenAPIDefinition;
import io.swagger.v3.oas.annotations.enums.SecuritySchemeType;
import io.swagger.v3.oas.annotations.info.Info;
import io.swagger.v3.oas.annotations.security.SecurityScheme;

import org.springframework.context.annotation.Configuration;

@Configuration
@OpenAPIDefinition(
    info = @Info(
        title = "Demo API",
        version = "1.0"
    )
)
@SecurityScheme(
    name = "Bearer Authentication",
    type = SecuritySchemeType.HTTP,
    scheme = "bearer",
    bearerFormat = "JWT"
)
public class OpenApiConfig {
}

package com.canadalife.demo.controller;

import org.springframework.security.access.prepost.PreAuthorize;
import org.springframework.web.bind.annotation.DeleteMapping;
import org.springframework.web.bind.annotation.GetMapping;
import org.springframework.web.bind.annotation.PostMapping;
import org.springframework.web.bind.annotation.PutMapping;
import org.springframework.web.bind.annotation.RequestMapping;
import org.springframework.web.bind.annotation.RestController;

import io.swagger.v3.oas.annotations.security.SecurityRequirement;

@RestController
@RequestMapping("/api/v2/product")
@SecurityRequirement(name = "Bearer Authentication")
public class ProductController {

	
	@PostMapping
	@PreAuthorize("hasAuthority('ADMIN')")
	public String addProducgt() {
		return "Product is added";
	}
	
	@PutMapping
	public String updateProduct() {
		return "Updated the product";
	}
	
	@GetMapping
	@PreAuthorize("hasAnyAuthority('USER','ADMIN')")
	public String getAllProducts() {
		return "All products are listed now";
	}
	
	@DeleteMapping
	public String deleteProduct() {
		return "Deleting the product";
	}
	
}

	public static void main(String []args){

the above was traditional approach.

lambda approach?


interface Person{
	void speak();
}

public class App{

	public static void main(String []args){
		Person raja=()->{
			System.out.println();
		}

		raja.speak();
	}
}


-------------------------------------------
the above was using lambda.



when i use lambda, i am providing the code.
But what if some other class already have suitable code, but does not implement Person interface. And may not named as "speak" method.

ex:


interface Person{
	void speak();
}


class Student{			//this class does not implement Person interface
	public void talk(){		//see this method name is not "speak"
		System.out.println("This student is speaking");
	}
}


public class App{

	public static void main(String []args){
		Person raja=new Student()::talk;
		raja.speak();
	}
}


---------------------------
the above is an example for method reference.


now, lets slightly modify the method as static.

interface Person{
	void speak();
}


class Student{			//this class does not implement Person interface
	public static void talk(){		//see this method name is not "speak"
		System.out.println("This student is speaking");
	}
}


public class App{

	public static void main(String []args){
		Person raja=Student::talk;			//static method reference. because, talk is static, and ref by class not object
		raja.speak();
	}
}
