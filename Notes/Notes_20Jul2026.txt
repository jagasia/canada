SELF JOIN
-----------


SELECT E.EMPLOYEE_ID, E.FIRST_NAME, E.MANAGER_ID, M.FIRST_NAME||' '||M.LAST_NAME
from HR.EMPLOYEES E
JOIN HR.EMPLOYEES M
ON E.MANAGER_ID=M.EMPLOYEE_ID
ORDER BY E.EMPLOYEE_ID;
